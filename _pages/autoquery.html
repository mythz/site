AutoQuery

The fastest way to servicify your existing database with intuitive typed end to end smart clients.

Present your data instantly with AutoGen

Connect your database and enable configure AutoQuery to use AutoGen to generate typed end to end HTTP services for your database tables. 

// Connect your database
container.AddSingleton<IDbConnectionFactory>(c =>
    new OrmLiteConnectionFactory(MapProjectPath("~/northwind.sqlite"), SqliteDialect.Provider));

// Configure AutoQuery to Generate CRUD services
Plugins.Add(new AutoQueryFeature {
    MaxLimit = 1000,
    GenerateCrudServices = new GenerateCrudServices {
        AutoRegister = true
    }
});

AutoGen in Action

AutoQuery services are ServiceStack services

Integrates with all core ServiceStack features and tools seamlessly for maximum reuse and composability.

Add tables using code-first with minimal code

// Connect your database
container.AddSingleton<IDbConnectionFactory>(c =>
    new OrmLiteConnectionFactory(MapProjectPath("~/northwind.sqlite"), SqliteDialect.Provider));

// Add the AutoQuery Plugin
Plugins.Add(new AutoQueryFeature { MaxLimit = 100 });

// Add your custom DTO with route to register a table to query.
[Route("/query/customers")]
public class QueryCustomers : QueryDb<Customer> {}

That's all you need, you now have a fully queryable, type safe API for the Customer table.

Customize further to fine tune your services

// Connect your database
container.AddSingleton<IDbConnectionFactory>(c =>
    new OrmLiteConnectionFactory(MapProjectPath("~/my_database.sqlite"),
      SqliteDialect.Provider));

// Add the AutoQuery Plugin
Plugins.Add(new AutoQueryFeature { MaxLimit = 100 });

// Return CustomRockstar result to control what the service returns
// Join with other tables
[Route("/rockstar-albums")]
public class QueryRockstarAlbums
  : QueryDb<Rockstar,CustomRockstar>, IJoin<Rockstar,RockstarAlbum>
{
    public int? Age { get; set; }
    public string RockstarAlbumName { get; set; }
}

// Custom result
public class CustomRockstar
{
    public string FirstName { get; set; }
    public string LastName { get; set; }
    public int? Age { get; set; }

    // Comes from joined table
    public string RockstarAlbumName { get; set; }
}

// Override with custom implementation
public class MyQueryServices : Service
{
    public IAutoQueryDb AutoQuery { get; set; }

    public async Task<object> Any(QueryRockstarAlbums query)
    {
        using var db = AutoQuery.GetDb(query, base.Request);
        var q = AutoQuery.CreateQuery(query, base.Request, db);
        return await AutoQuery.ExecuteAsync(query, q, base.Request, db);
    }
}
